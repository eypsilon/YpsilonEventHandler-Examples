// Implementation code: /** * Load multiple scripts from an array of URLs * @param {string[]} urls - Array of script URLs to load * @returns {Promise} - Promise that resolves when all scripts are loaded */ async function loadScripts(urls) { const results = []; const total = urls.length; // Update progress updateProgress(0, total); // Create all script loading promises const promises = urls.map((url, index) => { return new Promise((resolve) => { const script = document.createElement('script'); script.src = url; script.onload = () => { results[index] = { url, status: 'success' }; updateProgress(results.filter(r => r).length, total); resolve(); }; script.onerror = () => { results[index] = { url, status: 'error' }; updateProgress(results.filter(r => r).length, total); resolve(); }; document.head.appendChild(script); }); }); // Wait for all scripts to load (successfully or not) await Promise.all(promises); return results; } // Helper function to update progress function updateProgress(loaded, total) { const percent = total > 0 ? Math.round((loaded / total) * 100) : 0; document.getElementById('progress-fill').style.width = `${percent}%`; document.getElementById('progress-text').textContent = `${percent}% (${loaded}/${total} scripts loaded)`; }
